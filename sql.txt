/*
For a given table:
    Count total rows
    Count distinct values for each column
    Generate SQL dynamically so you don’t hand-code columns
*/
USE Bronze_FinOps_Entity
DECLARE @TableSchema SYSNAME = 'dbo';
DECLARE @TableName SYSNAME = 'assetlocationentity';
DECLARE @SQL NVARCHAR(MAX);

-- Build the dynamic SELECT list
WITH cols AS (
    SELECT 
        COLUMN_NAME
    FROM INFORMATION_SCHEMA.COLUMNS
    WHERE TABLE_SCHEMA = @TableSchema   
      AND TABLE_NAME = @TableName
      -- Optional filters:
      -- AND COLUMN_NAME <> 'RECID'
      -- AND COLUMN_NAME LIKE '%ID'
)

SELECT @SQL = 
    'SELECT ' +
    '''' + @TableName + ''' AS TABLE_NAME, ' +
    'COUNT(*) AS COUNT_ALL_RECORDS, ' +
    STRING_AGG(
        'COUNT(DISTINCT ' + QUOTENAME(COLUMN_NAME) + ') AS ' + QUOTENAME('COUNTDISTINCT_' + COLUMN_NAME),
        ', '
    ) +
    ' FROM ' + QUOTENAME(@TableSchema) + '.' + QUOTENAME(@TableName)
FROM cols
EXECUTE(@SQL)
